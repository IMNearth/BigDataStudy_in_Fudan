def sigmoid(x:np.ndarray)->np.ndarray:
    return 1/(1 + np.exp(-x))

def sigmoid_diff(x:np.ndarray)->np.ndarray:
    return sigmoid(x) * (1 - sigmoid(x))

def tanh_diff(x:np.ndarray)->np.ndarray:
    return 1 - np.power(np.tanh(x), 2)

def relu(x:np.ndarray)->np.ndarray:
    return (abs(x) + x) / 2

def softmax(x:np.ndarray)->np.ndarray:
    y = np.exp(x)
    return y / y.sum(axis=1).reshape(-1, 1)

class NumRecognizer(object):
    ''' Number Recognizer for MINST Dataset. '''
    ACT = {
        "sigmoid": sigmoid, 
        "tanh": np.tanh, 
        "relu": relu
    }
    ACT_DIFF = {
        "sigmoid": sigmoid_diff, 
        "tanh": tanh_diff, 
        "relu": None
    }

    def __init__(self, 
        hidden_size:int, input_size:int=28*28, out_size:int=10, act_name="relu"):

        self.num_layers = 2
        self.in_weight = np.random.randn(input_size+1, hidden_size)
        self.out_weight = np.random.randn(hidden_size+1, out_size)

        self.act_name = act_name
        self.activation = self.ACT[self.act_name]
        self.act_diff = self.ACT_DIFF[self.act_name]

        self.x, self.ip, self.fp = None, None, None

    def forward(self, x:np.ndarray)->np.ndarray:
        ''' Forward through the network 
        :param x: data with shape (batch_size, input_sie) '''
        pad_array = np.ones((x.shape[0], 1))
        x = np.concatenate((pad_array, x), axis=1)
        self.x = x      # batch_size x (input_size + 1)
        x = x @ self.in_weight
        self.ip = x     # batch_size x hidden_size
        
        x = self.activation(x)
        pad_array = np.ones((x.shape[0], 1))
        x = np.concatenate((pad_array, x), axis=1)
        self.fp = x     # batch_size x (hidden_size + 1)
        
        x = x @ self.out_weight
        return x
    
    def __call__(self, x):
        return self.forward(x)

    def nll_loss(self, logit_y, true_y):
        ''' implement the negative log-likelihood loss here '''
        loss = log_softmax(logit_y, axis=-1)
        loss = (-1) * loss[np.arange(len(true_y)), true_y].sum()
        return loss

    def step(self, logit_y, true_y, lr:float=0.001, reg_coef:float=0.0):
        ''' stochastic gradient descent 
        
        :param lr: learning rate
        :param reg_coef: l2 regularization coefficient
        '''
        batch_size = len(true_y)
        
        prob_y = softmax(logit_y)
        ground_y = np.zeros_like(prob_y)
        ground_y[np.arange(prob_y.shape[0]), true_y] = 1.0
        err = prob_y - ground_y     # batch_size x out_size

        grad_out = self.fp.transpose() @ err    # hidden_size+1 x out_size
        
        if self.act_diff is not None:
            diff_act = self.act_diff(self.ip)
        else:
            diff_act = np.zeros_like(self.ip)
            diff_act[self.ip.nonzero()] = 1.0
        
        backprop = diff_act * (err @ self.out_weight[1:].transpose())  # batch_size x hidden_size
        grad_in = self.x.transpose() @ backprop                        # input_size +1 x hidden_size

        # update
        self.out_weight = self.out_weight - \
            (lr * grad_out / batch_size + reg_coef * self.out_weight)
        self.in_weight = self.in_weight - \
            (lr * grad_in / batch_size + reg_coef * self.in_weight)
        
        self.x, self.ip, self.fp = None, None, None
        pass

    def save(self, path):
        pkl.dump(open(path, "wb"), [(self.in_weight, self.out_weight), self.act_name])
    
    def load(self, path):
        (self.in_weight, self.out_weight), self.act_name = pkl.load(open(path, "rb"))
        self.activation = self.ACT[self.act_name]
        self.act_diff = self.ACT_DIFF[self.act_name]
        return self

train_data, train_lab = pkl.load(open("data/MNIST/train.pkl", "rb"))
test_data, test_lab = pkl.load(open("data/MNIST/test.pkl", "rb"))
num_train_samp = train_data.shape[0]
num_test_samp = test_data.shape[0]

# reshape and standardize data
train_data = train_data * 1.0
train_data = train_data.reshape(num_train_samp, -1)
train_mean = train_data.mean()
train_data = train_data - train_mean

test_data = test_data * 1.0
test_data = test_data.reshape(num_test_samp, -1)
test_data = test_data - train_mean

# print
print("Train:", train_data.shape, train_lab.shape)
print("Test:", test_data.shape, test_lab.shape)


def train(hidden_size, batch_size, num_epochs, learn_rate, l2_rate, pt=True, lr_desc=False):

    random.seed(2022)
    np.random.seed(2022)
    
    # init model
    model = NumRecognizer(
        hidden_size=hidden_size,
        input_size=train_data.shape[-1],
        out_size=10, act_name="tanh"
    )

    # training settings
    num_iters = num_train_samp // batch_size
    if num_train_samp % batch_size !=0: num_iters += 1

    # start training
    train_indices = list(range(num_train_samp))
    train_losses, test_losses, test_accs = [], [], []
    for ep in range(num_epochs):
        random.shuffle(train_indices)

        epoch_loss = .0
        for it in range(num_iters):
            cur_ind = train_indices[it*batch_size: (it+1)*batch_size]
            cur_data, cur_lab = train_data[cur_ind], train_lab[cur_ind]
            
            logits = model(cur_data)
            cur_loss = model.nll_loss(logits, cur_lab)
            epoch_loss += cur_loss

            model.step(logits, cur_lab, lr=learn_rate, reg_coef=l2_rate)

            # if it % 10 == 0:
            #     train_losses.append(cur_loss)
                
            #     logits = model(test_data)
            #     test_loss = model.nll_loss(logits, test_lab)
            #     test_loss = test_loss
            #     test_losses.append(test_loss / num_test_samp * batch_size)
                
            #     acc = (logits.argmax(-1) == test_lab).sum() / num_test_samp
            #     test_accs.append(acc)
        
        if lr_desc and ep % 2 == 0:
            learn_rate = learn_rate * 0.9

        epoch_loss = epoch_loss / num_train_samp
        str_ = "Epoch {} with average loss: {:.3f}".format(ep, epoch_loss)
        if pt: print(str_)
    
    return model, (train_losses, test_losses, test_accs)


def test(model, ):
    logits = model(test_data)
    pred_lab = logits.argmax(-1)
    acc = (pred_lab == test_lab).sum() / num_test_samp

    return acc

def draw_hyparam_cmp(full_data):
    fig = plt.figure(figsize=(18, 5))
    
    for i, hidden_size in enumerate([64, 128, 256]):
        interest_data = full_data[full_data['hidden_size'] == hidden_size]

        X = interest_data["learn_rate"].astype(float)
        Y = interest_data["l2_rate"].astype(float)

        xi = np.array(sorted(list(set(X.tolist()))))
        yi = np.array(sorted(list(set(X.tolist()))))
        xig, yig = np.meshgrid(xi, yi)

        ax = fig.add_subplot(int(f"13{i+1}"), projection='3d')
        
        Z = interest_data['acc'].astype(float)
        zi = griddata((X, Y), Z, (xi[None,:], yi[:,None]), method='nearest')
        print(zi)
        ax.plot_surface(xig, yig, zi, rstride=1, cstride=1, cmap=cm.coolwarm)

        ax.xaxis.set_major_locator(MultipleLocator(0.005))
        ax.yaxis.set_major_locator(MultipleLocator(0.005))
        # ax.zaxis.set_major_locator(MultipleLocator(1.0))
        
        average_acc = interest_data['acc'].mean()   
        ax.set_xlim(xi[0], xi[-1])
        ax.set_ylim(yi[0], yi[-1])
        # ax.set_zlim(int(min(Z.min(), model_sr))-0.5, int(Z.max())+0.5)

        nxi = np.linspace(X.min()-0.1,X.max()+0.1,len(Z))
        nyi = np.linspace(Y.min()-0.15,Y.max()+0.15,len(Z))
        ax.plot(xs=nxi, ys=nyi*.0+average_acc, zs=nyi[-1], zdir='y', linestyle='--', color="r")
        ax.plot(xs=nyi, ys=nxi*.0+average_acc, zs=nxi[0], zdir='x', linestyle='--', color="r")

        #
        ax.tick_params(labelsize=14, pad=0)
        ax.set_xlabel('learning rate', fontsize=14)
        ax.set_ylabel('l2-penalty rate', fontsize=14)
        ax.set_zlabel('acc', fontsize=14)
        ax.set_title(f"hidden size = {hidden_size}", pad=15, fontsize=20)

    plt.tight_layout()

    return fig

def draw_learning_curve(loss_dict):
    ''' fixed / descent '''
    fig, axes = plt.subplots(1, 3, figsize=(18, 5), squeeze=True)

    line_styles = {"fixed": "-", "descent": "--"}
    colors = {"fixed": "r", "descent": "b"}

    # ax[0] - for train loss
    ax = axes[0]
    for lr_schema in loss_dict:
        train_loss, test_loss, __ = loss_dict[lr_schema]
        x = np.array(list(range(len(test_loss)))) * 10
        ax.plot(x, train_loss, color=colors[lr_schema], linestyle=line_styles[lr_schema], label=f"{lr_schema}")
    
    ax.tick_params(labelsize=14, pad=0)
    ax.set_ylim(10, 450)
    ax.set_xlim(1000, 5000)
    ax.xaxis.set_major_locator(MultipleLocator(1000))
    ax.yaxis.set_major_locator(MultipleLocator(100))
    ax.set_xlabel('', fontsize=14)
    ax.set_ylabel('', fontsize=14)
    ax.set_title('Train Loss', fontsize=20)
    ax.legend(loc="best", fontsize=14) # ncol=2

    # ax[1] - for test loss
    ax = axes[1]
    for lr_schema in loss_dict:
        train_loss, test_loss, __ = loss_dict[lr_schema]
        ax.plot(x, test_loss, color=colors[lr_schema], linestyle=line_styles[lr_schema], label=f"{lr_schema}")

    ax.tick_params(labelsize=14, pad=0)
    ax.set_ylim(10, 350)
    ax.set_xlim(1000, 5000)
    ax.xaxis.set_major_locator(MultipleLocator(1000))
    ax.yaxis.set_major_locator(MultipleLocator(100))
    ax.set_xlabel('iters', fontsize=14)
    ax.set_ylabel('', fontsize=14)
    ax.set_title('Test Loss', fontsize=20)
    ax.legend(loc="best", fontsize=14) # ncol=2

    # ax[2] - for test accuracy
    ax = axes[2]
    for lr_schema in loss_dict:
        __, __, test_acc = loss_dict[lr_schema]
        ax.plot(x, test_acc, color=colors[lr_schema], linestyle=line_styles[lr_schema], label=f"{lr_schema}")
    
    ax.tick_params(labelsize=14, pad=0)
    ax.set_ylim(0.75, 1)
    ax.set_xlim(1000, 5000)
    ax.xaxis.set_major_locator(MultipleLocator(1000))
    ax.yaxis.set_major_locator(MultipleLocator(0.1))
    ax.set_xlabel('', fontsize=14)
    ax.set_ylabel('', fontsize=14)
    ax.set_title('Test Accuracy', fontsize=20)
    ax.legend(loc="best", fontsize=14)

    plt.tight_layout()
    plt.show()

    return fig
